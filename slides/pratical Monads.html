// concat :: (Monad m) => m m x -> m x
[1].concat([2]) -> [1,2]
// concat Array Array -> Array
T concat T -> T
type G[X] = T[T[X]]
G = T compose T = T.T
T.T[X] = G[X] = T[T[X]]
concat(T.T).T == T.concat(T.T)
